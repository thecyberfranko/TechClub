JavaScript

About

Arguably the most used programming language, JavaScript was written by Brendan Eich in May 1995 for the Netscape Navigator 2.0 beta in 10 days. It was standardized in 1997 by Ecma International as the first ECMAScript, which today ECMAScript 2016 is the current standard. Interpreted and nearly supported by all modern web browsers, JavaScript is extremely portable, increasingly integrated into the web by enabling dynamic content and pages. Also, because JavaScript was designed for web designers and amateur programmers it is a good learning language.

Online/Installation

Because JavaScript is included in web browsers and it is likely already installed. These examples embed JavaScript into an html page by containing the JavaScript code between the <script> tags. To run save the code in a file with the .html extension and open it with the browser. Alternatively the JavaScript code saved in a ".js" file is incoporated to a html page with the <script type="text/javascript" src="Path/To/File.js"></script>.

    convertedPhrase = ""
    # Convert entire phrase to uppercase and split into separate words.

Factor.py


from math import sqrt
def factor(Num):
    factors, limitSearch = [], int(sqrt(Num)) + 1
    for Divisor in range(1, limitSearch):
        Whole, Remainder = divmod(Num, Divisor)
        if not Remainder:
            factors.append((Divisor, Whole))
    return factors

print(factor(48))

Factor.java


import java.util.ArrayList;

public class Factor
{
    public static void main(String[] args)
    {
        for (Integer[] Pair: factor(48)) {
            System.out.printf("%d, %d%n", Pair[0], Pair[1]);
        }
    }

    public static ArrayList<Integer[]> factor(int Num) {
        ArrayList<Integer[]> factors = new ArrayList<Integer[]>();
        int limitSearch = (int)Math.sqrt(Num) + 1;
        for (int Divisor = 1; Divisor < limitSearch; ++Divisor) {
            if (Num % Divisor == 0) {
                factors.add(new Integer[]{Divisor, Num / Divisor});
            }
        }
        return factors;
    }
}

PigLatin.cpp


#include <iostream>
#include <string>
#include <vector>
#include <cctype>

std::vector<std::string> tokenize(std::string initalPhrase, char token) {
    std::vector<std::string> Tokenized;
    std::string Word = "";
    for (char letter: initalPhrase) {
        if (letter != token) {
            Word += std::toupper(letter);
            continue;
        }
        Tokenized.push_back(Word);
        Word = "";
    }
    Tokenized.push_back(Word);
    return Tokenized;
}

std::string translateToPigLatin(std::string initalPhrase) {
    std::string convertedPhrase = "";
    for (std::string Word: tokenize(initalPhrase, ' ')) {
        if (Word.length() > 1) {
            convertedPhrase += Word.substr(1);
        }
        convertedPhrase += Word.substr(0, 1) + "AY ";
    }
    return convertedPhrase;
}

int main() {
    std::string initalPhrase = "This example converts this phrase to Pig Latin";
    std::cout << translateToPigLatin(initalPhrase) << std::endl;
}

Factor.cpp


#include <iostream>
#include <vector>
#include <cmath>
#include <utility>

std::vector<std::pair<int, int>> factor(int Num) {
    std::vector<std::pair<int, int>>  factors;
    int limitSearch = std::sqrt(Num) + 1;
    for (int Divisor = 1; Divisor < limitSearch; ++Divisor) {
        if (!(Num % Divisor)) {
            factors.push_back(std::make_pair(Divisor, Num / Divisor));
        }
    }
    return factors;
}

int main()
{
    for (std::pair<int, int> Pair: factor(48)) {
        std::cout << Pair.first << ", " << Pair.second << std::endl;
    }
    return 0;
}
